CREDITOS_TRANSACCIONES

CREATE TABLE CREDITOS_TRANSACCIONES (ID TLLAVE NOT NULL,
        FECHA TFECHAHORA NOT NULL,
        TIPO_MOVIMIENTO TCADENAMEDIANA NOT NULL,
        ABONO_ID TLLAVE NOT NULL,
        VENTA_ID TLLAVE NOT NULL,
        TURNO_ID TLLAVE NOT NULL,
        CLIENTE_ID TLLAVE NOT NULL,
        SALDO_AL_MOMENTO TMONEDA DEFAULT 0,
        MONTO_DEL_MOVIMIENTO TMONEDA DEFAULT 0,
        CAJERO_ID TLLAVE NOT NULL,
        CAJA_ID TLLAVE NOT NULL,
        DEVOLUCION_ID TLLAVE DEFAULT 0,
CONSTRAINT PK_CREDITOS_TRANSACCIONES_ID PRIMARY KEY (ID));
