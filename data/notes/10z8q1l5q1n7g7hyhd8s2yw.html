REPORTE_ABONO_SERVICIOS

CREATE TABLE REPORTE_ABONO_SERVICIOS (ID_TRANSACCION TGUID NOT NULL,
        BANCO TCADENAMEDIANA,
        CUENTA_ID TCADENAMEDIANA,
        FOLIO TCADENAMEDIANA,
        FOLIO_DE_PROVEEDOR TCADENAMEDIANA,
        FOLIO_CANCELACION TCADENAMEDIANA,
        MONTO TMONEDA NOT NULL,
        ABONADO_EN TFECHAHORA,
        CANCELADO_EN TFECHAHORA,
        MENSAJE TCADENALARGA,
        CUENTA_ORIGEN VARCHAR(50),
        OTRA_INFORMACION VARCHAR(1000),
        METODO_DE_PAGO VARCHAR(100),
        REPORTADO_PREVIAMENTE "BOOLEAN" DEFAULT 0,
CONSTRAINT PK_REPORTE_ABONO_SERVICIO PRIMARY KEY (ID_TRANSACCION));
