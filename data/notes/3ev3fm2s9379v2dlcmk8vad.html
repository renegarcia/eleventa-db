PRODUCTOS

CREATE TABLE PRODUCTOS (ID TLLAVE NOT NULL,
        CODIGO TCADENAMEDIANA NOT NULL,
        DESCRIPCION TPRODUCTODESCRIPCION,
        TVENTA TTIPOVENTA,
        PCOSTO TMONEDA,
        PVENTA TMONEDA,
        DEPT TLLAVE,
        PROVID TLLAVE,
        UMEDIDA TLLAVE,
        MAYOREO TMONEDA,
        IPRIORIDAD TSMALLINT,
        DINVENTARIO TCANTIDADINV,
        DINVMINIMO TCANTIDADINV,
        DINVMAXIMO TCANTIDADINV,
        CHECADO_EN TFECHA,
        PORCENTAJE_GANANCIA TPORCENTAJEGANANCIA DEFAULT 0,
        IMPUESTOS TCADENALARGA,
        PVENTA_ANTERIOR TMONEDA DEFAULT 0,
        PCOSTO_ANTERIOR TMONEDA DEFAULT 0,
        PMAYOREO_ANTERIOR TMONEDA DEFAULT 0,
        MEDIDA_ID TLLAVE,
        PFINAL TMONEDA DEFAULT 0,
        PMAYOREOFINAL TMONEDA DEFAULT 0,
        ES_KIT TBOOLEANO DEFAULT 'f',
        USA_INVENTARIO TBOOLEANO DEFAULT 'f',
        ELIMINADO_EN TFECHAHORA,
CONSTRAINT PK_PRODUCTOS PRIMARY KEY (ID),
CONSTRAINT UNQ1_PRODUCTOS UNIQUE (CODIGO));
