IMPUESTOS

CREATE TABLE IMPUESTOS (ID TLLAVE NOT NULL,
        NOMBRE TCADENAMEDIANA,
        PORCENTAJE TCANTIDAD,
        DEFECTO TBOOLEANO,
        ACTIVO TBOOLEANO,
        ORIGEN TCARACTER DEFAULT 'F' NOT NULL,
        BASE_GRAVABLE TBOOLEANO DEFAULT 'F' NOT NULL,
        TIPO CHAR(1) DEFAULT 'T' NOT NULL,
CONSTRAINT PK_IMPUESTOS PRIMARY KEY (ID));
