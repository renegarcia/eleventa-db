CORTE_OPERACIONES

CREATE TABLE CORTE_OPERACIONES (ID TLLAVE NOT NULL,
        DIA TFECHAHORA NOT NULL,
        ID_CAJA TLLAVE,
        ACUMULADO_VENTAS TCURRENCY,
        ACUMULADO_ENTRADAS TCURRENCY,
        ACUMULADO_SALIDAS TCURRENCY,
        ACUMULADO_GANANCIA TCURRENCY,
        VENTAS_EFECTIVO TCURRENCY,
        VENTAS_TARJETA TCURRENCY,
        VENTAS_VALES TCURRENCY,
        VENTAS_CREDITO TCURRENCY,
        ABONOS_EFECTIVO TCURRENCY,
        DEVOLUCIONES_VENTAS_EFECTIVO TCURRENCY,
        DEVOLUCIONES_VENTAS_CREDITO TCURRENCY,
        DEVOLUCIONES_ABONOS_EFECTIVO TCURRENCY,
        NUMERO_VENTAS TINTEGER DEFAULT 0,
        DEVOLUCIONES_VENTAS_VALES TCURRENCY,
        ABONOS_TARJETA TCURRENCY DEFAULT 0,
        ABONOS_VALES TCURRENCY DEFAULT 0,
        COMISIONES_TARJETA TMONEDA DEFAULT 0,
        DEVOLUCIONES_VENTAS_TARJETA NUMERIC(10, 4) DEFAULT 0.0000 NOT NULL,
CONSTRAINT PK_CORTE_OPERACIONES PRIMARY KEY (ID),
CONSTRAINT UNQ1_CORTE_OPERACIONES UNIQUE (DIA, ID_CAJA) USING INDEX IX_DIA_CAJA);
