CAJAS

CREATE TABLE CAJAS (ID TLLAVE NOT NULL,
        NOMBRE TCADENAMEDIANA,
        ULTIMA_IP TCADENAMEDIANA NOT NULL,
        ULTIMO_INGRESO TFECHAHORA,
        NOMBRE_PC TNOMBREPC,
        ELIMINADA_EN TFECHAHORA,
        ES_PRINCIPAL TBOOLEANO DEFAULT 'F',
        TERMINAL_ID VARCHAR(40),
CONSTRAINT PK_CAJAS PRIMARY KEY (ID),
CONSTRAINT UNQ1_CAJAS UNIQUE (NOMBRE, ELIMINADA_EN) USING INDEX IX_CAJA_NOMBRE_UNICO_ACTIVO,
CONSTRAINT UNQ1_CAJA_SID UNIQUE (NOMBRE_PC, ELIMINADA_EN));
