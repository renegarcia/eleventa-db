OPERACIONES

        VENTAS TMONEDA DEFAULT 0,
        SALIDAS TMONEDA DEFAULT 0,
        ENTRADAS TMONEDA DEFAULT 0,
        PAGOS TMONEDA DEFAULT 0,
        IMPUESTOS TMONEDA DEFAULT 0,
        GANANCIAS TMONEDA DEFAULT 0,
        ABONO_ID TLLAVE,
        INGRESOS_TARJETA TMONEDA DEFAULT 0,
