CREDITOS_ABONOS

CREATE TABLE CREDITOS_ABONOS (ID TLLAVE NOT NULL,
        FECHA TFECHAHORA NOT NULL,
        MONTO TMONEDA DEFAULT 0,
        NOTA TNOTA,
        FORMA_PAGO TFORMAPAGO,
        FACTURA_ID TLLAVE,
        CANCELADO_EN TFECHAHORA,
        OLD_ABONO_ID TLLAVE DEFAULT 0,
CONSTRAINT PK_CREDITOS_ABONOS_ID PRIMARY KEY (ID));
