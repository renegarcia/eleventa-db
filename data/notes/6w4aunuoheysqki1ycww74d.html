        CARRIER TCADENAMEDIANA NOT NULL,
        ENVIADA_EN TFECHAHORA NOT NULL,
        MONTO TSMALLINT NOT NULL,
        NUMERO_TELEFONO VARCHAR(21) NOT NULL,
        TRANSACCION_ID TNUMEROTRANSACCION NOT NULL,
        PROVEEDOR TCADENAMEDIANA NOT NULL,
        AUTORIZACION TCADENAMEDIANA,
        NO_ORDEN_PROVEEDOR TCADENAMEDIANA,
        VENTATICKET_ID TLLAVE,
        RECIBIDA_EN TFECHAHORA,
        FUE_EXITOSA TBOOLEANO DEFAULT 'F' NOT NULL,
        MENSAJE_FALLA TCADENALARGA,
        USUARIO TCADENAMEDIANA DEFAULT 0 NOT NULL,
CONSTRAINT PK_VENTA_RECARGA PRIMARY KEY (ID));
