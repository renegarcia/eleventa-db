MEDIDAS

CREATE TABLE MEDIDAS (ID TLLAVE NOT NULL,
        UNIDAD_ENTEROS TCADENAMEDIANA NOT NULL,
        UNIDAD_FRACCIONES TCADENAMEDIANA NOT NULL,
        FACTOR TINTEGER DEFAULT 1,
        ACTIVO TBOOLEANO DEFAULT 1,
CONSTRAINT PK_MEDIDAS PRIMARY KEY (ID),
CONSTRAINT UNQ1_MEDIDAS UNIQUE (UNIDAD_ENTEROS, ACTIVO));
