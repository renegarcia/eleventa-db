TURNOS

CREATE TABLE TURNOS (ID TLLAVE NOT NULL,
        ID_CAJA TLLAVE NOT NULL,
        ID_OPERACION_INICIO TLLAVE NOT NULL,
        ID_OPERACION_FIN TLLAVE,
        ID_CAJERO TLLAVE,
        INICIO_EN TFECHAHORA NOT NULL,
        TERMINO_EN TFECHAHORA,
        DINERO_INICIAL TCURRENCY NOT NULL,
        SOSPECHOSO CHAR(1) NOT NULL,
        ACUMULADO_VENTAS TCURRENCY NOT NULL,
        ACUMULADO_ENTRADAS TCURRENCY,
        ACUMULADO_SALIDAS TCURRENCY DEFAULT 0,
        ACUMULADO_GANANCIA TCURRENCY,
        VENTAS_EFECTIVO TCURRENCY,
        VENTAS_TARJETA TCURRENCY,
        VENTAS_VALES TCURRENCY,
        VENTAS_CREDITO TCURRENCY,
        EFECTIVO_AL_CIERRE TCURRENCY,
        ABONOS_EFECTIVO TCURRENCY,
        DEVOLUCIONES_VENTAS_EFECTIVO TCURRENCY,
        DEVOLUCIONES_VENTAS_CREDITO TCURRENCY,
        DEVOLUCIONES_ABONOS_EFECTIVO TCURRENCY,
        NUMERO_VENTAS TINTEGER,
        DEVOLUCIONES_VENTAS_VALES TCURRENCY,
        ABONOS_TARJETA TCURRENCY DEFAULT 0,
        ABONOS_VALES TCURRENCY DEFAULT 0,
        COMISIONES_TARJETA TMONEDA DEFAULT 0,
        SINCRONIZADO_EN TFECHAHORA,
        DEVOLUCIONES_VENTAS_TARJETA NUMERIC(10, 4) DEFAULT 0.0000 NOT NULL,
CONSTRAINT PK_TURNOS PRIMARY KEY (ID),
CONSTRAINT UNQ_TURNO_ABIERTO_CAJA UNIQUE (ID_CAJA, TERMINO_EN));
