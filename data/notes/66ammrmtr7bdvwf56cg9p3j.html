INVENTARIO_HISTORIAL

        CANTIDAD TCANTIDAD NOT NULL,
        DESCRIPCION TCADENALARGA,
        COSTO_UNITARIO TMONEDA,
        COSTO_DESPUES TMONEDA,
        AJUSTE_ID TLLAVE,
        RECIBO_INVENTARIO_ID TLLAVE,
        VENTA_ID TLLAVE,
        TRANSFERENCIA_ID TLLAVE,
        CAJA_ID TLLAVE,
        USUARIO_ID TLLAVE,
        ALMACEN_ID TLLAVE,
        VENTA_POR_KIT TBOOLEANO DEFAULT 'f',
PRIMARY KEY (ID));
